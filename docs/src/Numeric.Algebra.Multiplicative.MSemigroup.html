-- | Defines a multiplicative semigroup.
--
-- @since 0.1
type MSemigroup :: Type -> Constraint
class MSemigroup s where
  -- | @since 0.1
  (.*.) :: s -> s -> s

infixl 7 .*.

-- | @since 0.1
instance MSemigroup Double where
  .*. :: Double -> Double -> Double
(.*.) = Double -> Double -> Double
forall a. Num a => a -> a -> a
(*)
  {-# INLINE (.*.) #-}
